progress.mjs
#!/bin/env node
// // 打印彩色日志
import chalk from 'chalk';
console.log(chalk.yellow('黄色'));
// // 进度条库
var ProgressBar = await import('progress');
// // 指定进度条头部，完成表示，未完成标志，总数
var bar = new ProgressBar.default('[ :bar ] :current/:total', {
  head: '🌛',
  complete: '🐶',
  total: 10,
});
var timer = setInterval(function () {
  // 进度条增加进度
  bar.tick();
  if (bar.curr === 1) {
    // 不同进度渲染不同的颜色
    bar.interrupt('\x1b[31m');
  } else if (bar.curr === 5) {
    bar.interrupt('\x1b[33m');
  } else if (bar.curr === 8) {
    bar.interrupt('\x1b[32m');
  } else if (bar.complete) {
    // 进度条完成状态
    console.log('\ncompleted\n');
    clearInterval(timer);
  }
}, 100);
